## Code

```nasm
	AREA RESET, DATA, READONLY
	EXPORT __Vectors

__Vectors
	DCD 0X40001000
	DCD Reset_Handler
	ALIGN
	AREA mycode, CODE, READONLY
	ENTRY
	EXPORT Reset_Handler

Reset_Handler
	LDR R0, =NUM
	LDR R3, =RESULT
	LDRB R1, [R0]
	AND R2, R1, #0X0F
	CMP R2, #09
	BLS DOWN
	ADD R2, #07

DOWN
	ADD R2, #0X30
	STRB R2, [R3]
	AND R4, R1, #0XF0
	MOV R4, R4, LSR#04
	CMP R4, #09
	BLS DOWNER
	ADD R3, #07

DOWNER
	ADD R4, #0X30
	STRB R4, [R3, #01]

NUM DCD 0X0000003A
	AREA	data, DATA, READWRITE
RESULT DCD 0

	END
```
