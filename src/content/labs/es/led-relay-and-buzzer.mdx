```c
#include <LPC17xx.h>

int main(void)
{
	LPC_PINCON->PINSEL1 = 0x00000000;
	LPC_GPIO0->FIODIR = 3<<24;

	while(1)
	{
		if(!(LPC_GPIO2->FIOPIN & 1<<11))
			LPC_GPIO0->FIOSET = 3<<24;
		else
			LPC_GPIO0->FIOCLR = 3<<24;
	}
}
```
